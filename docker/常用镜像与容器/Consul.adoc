== 单机

[source,sh]
----
mkdir -p ~/opt/consul/
----

.~/opt/consul/stack.yml
[source,yaml,%linenums]
----
services:
  consul:
    image: hashicorp/consul:1.22
    hostname: consul
    container_name: consul
    ports:
      - "8300:8300"
      - "8500:8500"
    environment:
      - CONSUL_BIND_INTERFACE=eth0
    volumes:
      - data:/consul/data
    command:
      # https://developer.hashicorp.com/consul/commands/agent
      agent -server -ui -ui-content-path /consul -client 0.0.0.0
    restart: unless-stopped

volumes:
  data:

networks:
  default:
    external: true
    name: rebue
----

* 部署

[source,shell]
----
docker compose -f ~/opt/consul/stack.yml up -d
----


== Swarm

[source,sh]
----
mkdir -p ~/opt/consul/template/
----

.~/opt/consul/stack.yml
[source,yaml,%linenums]
----
services:
  svr:
    image: hashicorp/consul:1.22
    hostname: consul
    ports:
      #- "8300:8300"
      - "8500:8500"
    environment:
      - CONSUL_BIND_INTERFACE=eth0
      - CONSUL_CLIENT_INTERFACE=eth0
    volumes:
      - consuldata:/consul/data
    command:
      # https://developer.hashicorp.com/consul/commands/agent
      agent -server -ui -ui-content-path /consul -client 0.0.0.0
    deploy:
      placement:
        constraints:
          # 部署的节点指定是consul主机
          - node.hostname==consul
    logging:
      options:
        max-size: 8m

volumes:
  consuldata:

networks:
  default:
    external: true
    name: rebue
----

[source,bash]
----
docker stack deploy -c ~/opt/consul/stack.yml consul
----

.~/opt/consul/template/stack.yml
[source,yaml,%linenums]
----
services:
  template:
    image: nnzbz/consul-template:0.41
    volumes:
      - ~/opt/consul/template:/template
    command: ["-config=/template/deregister.hcl", "-template=/template/deregister.tpl:/template/deregister.sh"]
    deploy:
      placement:
        constraints:
          # 部署的节点指定是consul主机
          - node.hostname==consul
    logging:
      options:
        max-size: 8m

networks:
  default:
    external: true
    name: rebue
----

.~/opt/consul/template/deregister.hcl
[,json]
----
consul {
  address = "consul:8500"
}

template {
  source = "/template/deregister.tpl"
  destination = "/template/deregister.sh"
  command = "/template/deregister.sh"
  perms = 0775
  wait {
    min = "30s"
    max = "60s"
  }
}
----

.~/opt/consul/template/deregister.tpl
[source,bash]
----
#!/bin/sh
{{range services}}
{{range service .Name "any"}}
#{{.ID}} {{.Name}} {{.Status}}
{{if eq .Status "critical"}}
curl -X PUT "http://consul:8500/v1/agent/service/deregister/{{.ID}}"
{{end}}
{{end}}
{{end}}
----


[source,bash]
----
docker stack deploy -c ~/opt/consul/template/stack.yml consul
----

== 停止运行超过168小时(7天)会启动不起来
. 报错信息
+
[,shell]
----
refusing to rejoin cluster because server has been offline for more than the configured server_rejoin_age_max (168h0m0s) - consider wiping your data dir
----
. 解决方案
+
在配置文件中配置server_rejoin_age_max超过当前的停机时间
+
.~/opt/consul/consul.hcl
[source,hcl]
----
server_rejoin_age_max = "720000h"  # 允许300天离线
----
+
.~/opt/consul/stack.yml
[source,hcl]
----
services:
  consul:
    image: hashicorp/consul:1.22
    hostname: consul
    container_name: consul
    ports:
      - "8300:8300"
      - "8500:8500"
    environment:
      - CONSUL_BIND_INTERFACE=eth0
    volumes:
      - ~/opt/consul/consul.hcl:/consul/config/consul.hcl
      - data:/consul/data
    command:
      # https://developer.hashicorp.com/consul/commands/agent
      agent -server -ui -client 0.0.0.0 -config-file=/consul/config/consul.hcl
    restart: unless-stopped

volumes:
  data:
----
