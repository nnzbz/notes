== 1. docker compose

. 准备目录和文件
+
[source,sh]
----
mkdir -p ~/opt/mysql/data
vim ~/opt/mysql/docker-compose.yml
----

. docker compose文件
+
.~/opt/mysql/docker-compose.yml
[source,yaml]
----
services:
  mysql:
    image: mysql:9
    container_name: mysql
    hostname: mysql
    ports:
      - "3306:3306"
      - "33060:33060"
    environment:
      - MYSQL_ROOT_PASSWORD=root
      - TZ=CST-8
    command:
      --character-set-server=utf8mb4
      --collation-server=utf8mb4_unicode_ci
      --default-time-zone='+8:00'
      --max_connections=5000
      --skip-name-resolve
    volumes:
      - ~/opt/mysql/data:/var/lib/mysql
    logging:
      options:
        max-size: 8m
    restart: unless-stopped

networks:
  default:
    external: true
    name: rebue
----

. 启动
+
[source,sh]
----
docker compose -f ~/opt/mysql/docker-compose.yml up -d
----
